1function [ANGLE_OUT,RANGE_OUT] = LIDAR(VP, HA, LASER_RANGE, ANGULAR_RES, MAP)
 2
 3% Simulated LIDAR
 4
 5% make sure the position of VP is feasible before calling this function
 6
 7% VP: vehicle position
 8% HA: vehicle heading angle
 9% ANGULAR_RES Unit: Degree
10% ANGLE       Unit: Radian
11
12ANGULAR_RES     = ANGULAR_RES*pi/180;
13ANGLE           = 0:ANGULAR_RES:(2*pi-ANGULAR_RES);
14RANGE           = zeros(size(ANGLE));
15
16for i = 1:length(ANGLE)
17    X_RAY = VP(1) + [0; LASER_RANGE*cos(ANGLE(i))];
18    Y_RAY = VP(2) + [0; LASER_RANGE*sin(ANGLE(i))];
19    
20    % intersection points
21    MAX_ITS = 100; %#THLD# maximum number of intersection points
22    X_ITS   = NaN*ones(MAX_ITS, 1);
23    Y_ITS   = NaN*ones(MAX_ITS, 1);
24    CNT     = 0;
25    
26    % [XI, YI] = intersections(X_RAY, Y_RAY, MAP.BDY(:, 1), MAP.BDY(:, 2));
27    [XI, YI] = intersections_coder(X_RAY, Y_RAY, MAP.BDY(:, 1), MAP.BDY(:, 2));
28    
29    for k = 1:length(XI)
30        CNT = CNT + 1;
31        X_ITS(CNT) = XI(k);
32        Y_ITS(CNT) = YI(k);
33    end
34    
35    if ~isempty(MAP.OBS(1).PTS)
36        for j = 1:length(MAP.OBS)
37            % [XI, YI] = intersections(X_RAY, Y_RAY, MAP.OBS(j).PTS(:, 1), MAP.OBS(j).PTS(:, 2));
38            [XI, YI] = intersections_coder(X_RAY, Y_RAY, MAP.OBS(j).PTS(:, 1), MAP.OBS(j).PTS(:, 2));
39            
40            for k = 1:length(XI)
41                CNT = CNT + 1;
42                X_ITS(CNT) = XI(k);
43                Y_ITS(CNT) = YI(k);
44            end
45        end
46    end
47
48    X_ITS = X_ITS(1:CNT);
49    Y_ITS = Y_ITS(1:CNT);
50    
51    if CNT > 0
52        
53        DIST = zeros(1, length(X_ITS));
54        
55        for j = 1:length(X_ITS)
56            DIST(j) = norm([X_ITS(j) - VP(1), Y_ITS(j) - VP(2)]);
57        end
58        
59        RANGE(i) = min(DIST);
60    else
61        RANGE(i) = LASER_RANGE;
62    end
63end
64
65% including the heading angle
66% only the front of the vehicle is scanned
67% START_ANGLE range should be 0 to 360-ANGULAR_RES
68START_ANGLE     = HA - pi/2;
69START_ANGLE     = mod(START_ANGLE, pi*2);
70
71% range of START_IDX should be 1 to 360/ANGULAR_RES
72START_IDX       = floor(START_ANGLE/ANGULAR_RES) + 1;
73
74IDX_180         = pi/ANGULAR_RES;
75IDX_360         = 2*pi/ANGULAR_RES;
76
77if START_IDX >= 1 && START_IDX <= IDX_180
78    IDX_OUT     = START_IDX:1:START_IDX+IDX_180;
79    ANGLE_OUT   = (IDX_OUT-1)*ANGULAR_RES;
80    RANGE_OUT   = RANGE(IDX_OUT);
81elseif START_IDX > IDX_180;
82    IDX_OUT     = [START_IDX:1:IDX_360,1:1:(START_IDX-IDX_180)];
83    ANGLE_OUT   = (IDX_OUT-1)*ANGULAR_RES;
84    RANGE_OUT   = RANGE(IDX_OUT);
85else
86    ANGLE_OUT   = [];
87    RANGE_OUT   = [];
88end
89
90% the output is in vehicle's local coordinates
91ANGLE_OUT       = ANGLE_OUT - HA;
92ANGLE_OUT       = mod(ANGLE_OUT, pi*2);
